88      protected static boolean getParams(String[] args) {
89          if (args.length < 1) {
90              logger.error(Messages.getString("Configuration.NeedConfig")); //$NON-NLS-1$
91              JFileChooser chooser = new JFileChooser();
92              int returnvval = chooser.showOpenDialog(null);
93              if (returnvval == JFileChooser.APPROVE_OPTION) {
94                  File file = chooser.getSelectedFile();
95                  args = new String[] { file.getAbsolutePath() };
96              } else {
97                  JOptionPane.showMessageDialog(null, Messages.getString("Configuration.NeedConfig")); //$NON-NLS-1$
98                  return false;
99              }
100         }
101         if (!FileBasedConfiguration
102                 .setClientConfigurationFromXml(Configuration.configuration, args[0])) {
103             logger.error(Messages.getString("Configuration.NeedConfig")); //$NON-NLS-1$
104             return false;
105         }
106         Configuration.configuration.pipelineInit();
107         getEnvironnement().networkTransaction = new NetworkTransaction();
108         return true;
109     }

128         String[] args2;
129 
130         String pwd_gpp = Configuration.configuration.getServerKeyFile();
131         if (pwd_gpp == null) {
132             args2 = new String[] {
133                     "-ki",
134                     Configuration.configuration.getCryptoFile(),
135                     "-pwd",
136                     "testpassword",
137                     "-clear"
138             };
139         } else {
140             args2 = new String[] {
141                     "-ki",
142                     Configuration.configuration.getCryptoFile(),
143                     "-pi",
144                     pwd_gpp,
145                     "-pwd",
146                     "testpassword",
147                     "-clear"
148             };
149         }

213         btnCheckPartners = new JButton("Check Partners");
214         btnCheckPartners.addActionListener(new ActionListener() {
215             public void actionPerformed(ActionEvent arg0) {
216                 // Check all Known Partners
217                 try {
218                     TestPacket packet;
219                     String myhost = null;
220                     try {
221                         myhost = (AdminGui.getEnvironnement().hostId == null ?
222                                 InetAddress.getLocalHost().getHostName() : AdminGui.getEnvironnement().hostId);
223                     } catch (UnknownHostException e) {
224                         myhost = Messages.getString("AdminGui.NameUnknown");
225                     }
226                     packet = new TestPacket("MSG", "Administrator checking from " + myhost
227                             , 100);
228                     packet.retain();
229                     String result = Messages.getString("AdminGui.CheckedHosts");
230                     DbSession session = DbConstant.admin != null ? DbConstant.admin.getSession() : null;
231                     for (DbHostAuth host : DbHostAuth.getAllHosts()) {
232                         R66Future future = new R66Future(true);
233                         Message mesg = new Message(AdminGui.getEnvironnement().networkTransaction, future, host, packet);
234                         mesg.run();
235                         packet.retain();
236                         future.awaitUninterruptibly();
237                         if (future.isSuccess()) {
238                             result += "OK: " + host.toString() + "\n";
239                         } else {
240                             result += "KO: " + host.toString() + "\n";
241                         }
242                     }
243                     packet.clear();
244                     ;
245                     JOptionPane.showMessageDialog(null, result);
246                 } catch (WaarpDatabaseNoConnectionException e) {
247                     // TODO Auto-generated catch block
248                     e.printStackTrace();
249                 }
250             }
251         });

379     /**
380      * Quit application
381      */
382     private void quit() {
383         List<AdminXample> list = new ArrayList<AdminXample>();
384         for (AdminXample xample : xamples) {
385             if (xample.isStillLaunched()) {
386                 list.add(xample);
387             }
388         }
389         for (AdminXample xample : list) {
390             xample.exit();
391         }
392         List<AdminUiPassword> list2 = new ArrayList<AdminUiPassword>();
393         for (AdminUiPassword pwd : passwords) {
394             list2.add(pwd);
395         }
396         for (AdminUiPassword pwd : list2) {
397             pwd.exit(null);
398         }
399         if (AdminSimpleR66ClientGui.window != null) {
400             AdminSimpleR66ClientGui.window.frmRClientGui.dispose();
401             AdminSimpleR66ClientGui.window = null;
402         }
403         frmWaarpRCentral.dispose();
404         System.exit(0);
405     }
